92 def _make_cache_key(prefix: str, path: str, params: dict) -> str: 

93 """ 

94 Создает уникальный безопасный ключ для кэша: 

95 hashlib - стандартная библиотека Python для криптографических хешей; 

96 md5 - берёт любой объём данных и возвращает фиксированную строку длиной 32 символа в виде бинарного объекта; 

97 hexdigest - преобразует бинарный объект в читаемую строку 

98 """ 

99 raw = json.dumps(params, sort_keys=True, ensure_ascii=False).encode( 

100 "utf-8" 

101 ) # одинаковый порядок элементов словаря и правильная кодировка символов 

102 digest = hashlib.md5(raw).hexdigest() 

103 return f"tmdb_{prefix}:{path}:{digest}"[:200] # укорачиваем

105 def _get(self, path: str, params: dict | None = None, ttl_key: str = "recommended", retries=3, timeout=5) -> dict: 

106 """ 

107 Внутренний метод для GET запросов: 

108 timeout: 5 сек (защита от зависания) 

109 retries: 3 попытки 

110 Берет данные из кэша или кэширует (TTL: 1 час) 

111 """ 

112 url = f"{self._base_url}{path}" 

113 params = {**self._base_params, **(params or {})} 

114 

115 cache_key = self._make_cache_key("tmdb", path, params) # создаем уникальный кэш-ключ 

116 cached = cache.get(cache_key) # берем из кэша, если есть 

117 if cached is not None: 

118 return cached 

119 

120 for attempt in range(1, retries + 1): 

121 try: 

122 response = requests.get(url, params=params, timeout=timeout) 

123 response.raise_for_status() 

124 data = response.json() 

125 

126 cache.set(cache_key, data, TMDB_TTL.get(ttl_key, 60 * 60 * 12)) # по умолчанию кэширем на 12 часов 

127 return data 

128 

129 except (requests.exceptions.Timeout, requests.exceptions.ConnectionError): 

130 if attempt < retries: 

131 time.sleep(2 ** (attempt - 1)) # Backoff: 1s → 2s → 4s 

132 continue 

133 return {} 

134 except requests.exceptions.HTTPError as e: 

135 status = e.response.status_code 

136 if status in (429, 500, 502, 503, 504): 

137 if attempt < retries: 

138 time.sleep(2 ** (attempt - 1)) 

139 continue 

140 return {} 

141 return {} 

142 except JSONDecodeError: 

143 return {} 

144 return {}

146 def _get_multipage(self, path: str, pages: int = 1, params: dict = None, ttl_key: str = "recommended") -> list: 

147 """Возвращает несколько страниц результатов""" 

148 all_results = [] 

149 params = params or {} 

150 

151 first = self._get(path, {**params, "page": 1}, ttl_key) 

152 if "results" not in first: 

153 return [] 

154 

155 all_results.extend(first["results"]) 

156 total_pages = first.get("total_pages", 1) 

157 pages_to_fetch = min(pages, total_pages) 

158 

159 for page in range(2, pages_to_fetch + 1): 

160 resp = self._get(path, {**params, "page": page}, ttl_key) 

161 if "results" in resp: 

162 all_results.extend(resp["results"]) 

163 else: 

164 break 

165 return all_results
